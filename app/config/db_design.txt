# ユーザ, RA営業メンバー
;------------------------------------------------------------------------------

{
	user_id,					bigint, pk, ai,					# ユーザ
	login_id,					varchar, 32, aA0_, must,		# ログインID
	login_pw,					password, 32, aA0_				# ログインPW
	name,						varchar, 32, 1:30,				# 氏名
	email,						mail, must,						# メールアドレス

	deleted,					boolean, del,					# 削除
}
